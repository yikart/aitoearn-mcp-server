## 需要用到的几个请求
1. 获取账号列表
```js
var axios = require('axios');

var config = {
   method: 'get',
   url: 'http://127.0.0.1:7001/plugin/account/list',
   headers: { 
      'sk-key': 'sk-MW4r647ICeKqHCbjrUVdOcznaC7fUXH4'
   }
};

axios(config)
.then(function (response) {
   console.log(JSON.stringify(response.data));
})
.catch(function (error) {
   console.log(error);
});
```
2. 进行某个账号的发布
```js
var axios = require('axios');
var data = JSON.stringify({
   "skKey": "sk-MW4r647ICeKqHCbjrUVdOcznaC7fUXH4",
   "title": "美好生活",
   "desc": "记录美好生活",
   "type": "video",
   "videoUrl": "https://aitoearn.s3.ap-southeast-1.amazonaws.com/test/video.mp4",
   "coverUrl": "https://aitoearn.s3.ap-southeast-1.amazonaws.com/test/cover.jpg",
   "imgUrlList": null,
   "publishTime": "2025-08-08 20:30:00",
   "topics": "美好生活,记录"
});

var config = {
   method: 'post',
   url: 'http://127.0.0.1:7001/plugin/publish/create',
   headers: { 
      'sk-key': 'sk-MW4r647ICeKqHCbjrUVdOcznaC7fUXH4', 
      'Content-Type': 'application/json'
   },
   data : data
};

axios(config)
.then(function (response) {
   console.log(JSON.stringify(response.data));
})
.catch(function (error) {
   console.log(error);
});
```
